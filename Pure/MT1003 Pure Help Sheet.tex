    \subsection{Directed Graphs}
        Consists of set \(V\) of \emph{vertices} and set \(E\) of ordered pairs from \(V\) making \emph{edges}.
        Written: \(\Gamma = (V,E)\).\\
        Let \(\Gamma\) be a directed graph:
        \begin{description}
            \setlength{\itemsep}{0pt}
            \item[Walk] -- Sequence of verticies and edges on \(\Gamma\).
            \item[Path] -- Walk on \(\Gamma\) {\st} no vertex occurs twice.
            \item[Circuit] -- Closed walk on \(\Gamma\).
            \item[Simple Circuit] -- Closed path on \(\Gamma\).
            \item[Loop] -- Edge of form \((v,v)\) with \(v\in V\).
        \end{description}
    \subsection{Types \&  Properties of Graphs}
        \begin{description}
            \setlength{\itemsep}{0pt}
            \item[Undirected Graph] -- A graph {\st} \(\forall\ (v_i,v_j)\in\Gamma\ \exists\ (v_j,v_i)\). %
                Usually just called \emph{graph}.
            \item[Multigraph] --  Graph with multiple edges between verticies.
            \item[Simple Graph] -- Undirected, finite, no loops, no multiple edges.
            \item[Connected Graph] -- \(\exists\) path between any two distinct verticies \(a,b\). 
            \item[Vertex Degree] -- \# of edges incedent on vertex.
            \item[Regular Graph] -- All verticies have same degree.
            \item[Complete Graph] -- Graph, \(K_n\), is simple, connected with \(n\) verticies.
            \item[Null Graph] -- Graph, \(N_n\), with \(n\) verticies and no edges. Also a regular graph.
            \item[Cycle Length] -- Graph \(C_n\) {\st} it forms a circle.   
        \end{description}
    \subsection{Trees}
        \begin{description}
            \setlength{\itemsep}{0pt}
            \item[Tree] -- Connected, no circuits.
            \item[Spanning Tree] -- Subgraph of \(\Gamma\) which contains all verticies and is a tree. Minimal connectivity.
        \end{description}

    \subsection{Eulerian \& Hamiltonian Graphs}
        \begin{description}
            \setlength{\itemsep}{0pt}
            \item[Eulerian Graph] -- \(\exists\) circuit in \(\Gamma\) passing though every vertex and edge exactly once.
            \item[Semi-Eulerian Graph] -- \(\exists\) walk in \(\Gamma\) passing though every vertex and edge exactly once. 
        \end{description}
