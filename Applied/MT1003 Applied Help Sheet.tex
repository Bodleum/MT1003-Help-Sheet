  \subsection{\nth{2} Order \((L4)\)}
    {\bfseries Form:} \(A_{n+2}+bA_{n+1}+cA_{n}=F(n)\) with \(A_0=\alpha,\ A_1=\beta,\ b,c,\alpha,\beta\in\mathbb{R}\)\\
    \((A_n)_{CF}\) has sol'n determined by roots of \(p(x)=x^2+bx+c\):
    \begin{description}
      \item[Distinct, Real Roots:] \(x_{1,2}=\lambda_1,\lambda_2\to A_n=k_1\lambda_1^n+k_2\lambda_2^n\)
      \item[Repeated Roots:] \(x_{1,2}=\lambda\to A_n=\lambda^n(k_1+k_2n)\)
      \item[Complex Roots:] \(x_{1,2}=p\pm qi=re^{\pm i\vartheta}\to A_n=r^n\left(k_1\cos{(n\vartheta)}+k_2\sin{(n\vartheta)}\right)\) 
    \end{description}
